        /**
         * Returns the relevant IUCN Red List Category image given the RDB category. Not currently implemented,
         * @param iucnCategory
         * @param size
         * @return 
         * 
         */        
        public static function getIUCNCategoryImage(iucnCategory:String,size:String):Class
        {
            switch (iucnCategory)
            {
                case "CR":
                {
                    if (size=="s")
                    {
                        return iucncat_CR_sml;    
                    }
                    else
                    {
                        return iucncat_CR;
                    }
                    break;
                }
                case "LC":
                {
                    if (size=="s")
                    {
                        return iucncat_LC_sml;    
                    }
                    else
                    {
                        return iucncat_LC;
                    }
                    break;
                }
                case "DD":
                {
                    if (size=="s")
                    {
                        return null;    
                    }
                    else
                    {
                        return iucncat_DD;
                    }
                    break;
                }
                case "EN":
                {
                    if (size=="s")
                    {
                        return iucncat_EN_sml;    
                    }
                    else
                    {
                        return iucncat_EN;
                    }
                    break;
                }
                case "EW":
                {
                    if (size=="s")
                    {
                        return null;    
                    }
                    else
                    {
                        return iucncat_EW;
                    }
                    break;
                }
                case "EX":
                {
                    if (size=="s")
                    {
                        return null;    
                    }
                    else
                    {
                        return iucncat_EX;
                    }
                    break;
                }
                case "NE":
                {
                    if (size=="s")
                    {
                        return null;    
                    }
                    else
                    {
                        return iucncat_NE;
                    }
                    break;
                }
                case "NT":
                {
                    if (size=="s")
                    {
                        return iucncat_NT_sml;    
                    }
                    else
                    {
                        return iucncat_NT;
                    }
                    break;
                }
                case "VU":
                {
                    if (size=="s")
                    {
                        return iucncat_VU_sml;    
                    }
                    else
                    {
                        return iucncat_VU;
                    }
                    break;
                }
                default:
                {
                    return null;
                }
            }
        }
